    * Standard_List *


    insert(list, pos, item) : Add item to the position
    
    insert_last(list, item) : Add item to the last
    
    insert_first(list, item) : Add item to the first
    
    delete(list, pos) : delete item to the position
    
    clear(list) : delte all item in the list
    
    get_entry(list, pos) : return item in the position
    
    get_length(list) : get length of list
    
    is_empty(list) : check list is empty or not
    
    is_full(list) : check list is full or not
    
    print_list(list) : show all of the items in the list
